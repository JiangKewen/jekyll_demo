  // $& 表示匹配的字符串，即`b`本身
  // 所以返回结果与原字符串一致
  'abbc'.replaceAll('b', '$&')
  // 'abbc'

  // $` 表示匹配结果之前的字符串
  // 对于第一个`b`，$` 指代`a`
  // 对于第二个`b`，$` 指代`ab`
  'abbc'.replaceAll('b', '$`')
  // 'aaabc'

  // $' 表示匹配结果之后的字符串
  // 对于第一个`b`，$' 指代`bc`
  // 对于第二个`b`，$' 指代`c`
  'abbc'.replaceAll('b', `$'`)
  // 'abccc'

  // $1 表示正则表达式的第一个组匹配，指代`ab`
  // $2 表示正则表达式的第二个组匹配，指代`bc`
  'abbc'.replaceAll(/(ab)(bc)/g, '$2$1')
  // 'bcab'

  // $$ 指代 $
  'abc'.replaceAll('b', '$$')
  // 'a$c'
